def remove_outliers_iqr(series):
    Q1 = series.quantile(0.25)
    Q3 = series.quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    return series[(series >= lower_bound) & (series <= upper_bound)]

# Prepare an empty dict to store median compensation after outlier removal
median_comp_cleaned = {}

for lang, group in df_exploded.groupby('LanguageHaveWorkedWith'):
    comp_no_outliers = remove_outliers_iqr(group['ConvertedCompYearly'])
    if len(comp_no_outliers) > 5:  # Only consider languages with >5 samples after cleaning
        median_comp_cleaned[lang] = comp_no_outliers.median()

# Convert to Series and get top 10
median_comp_cleaned = pd.Series(median_comp_cleaned).sort_values(ascending=False).head(10)
